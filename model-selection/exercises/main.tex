\(x \sim \gauss(0, I_n)\)
  \begin{align*}
  y = Lx + a \implies y = Lx + a \sim \gauss(a, L^T L)
\end{align*}

\begin{align*}
  \E \left[ y \right]
  &=
  \E \left[ Lx + a \right] \\
  &=
  L \underbrace{\E \left[ x \right]}_{=0} + a \\
  &=
  a \\
\end{align*}

\begin{align*}
  \cov \left[ y \right]
  &=
  \cov \left[ Lx + a \right] \\
  &=
  L^T \underbrace{\cov \left[ x \right]}_{=I_n} L \\
  &=
  L^T L \\
\end{align*}
